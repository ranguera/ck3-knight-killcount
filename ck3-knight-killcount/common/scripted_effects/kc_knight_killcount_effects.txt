# Knight Kill Count - scripted effects
#
# kc_increment_kill_count_effect: run in the scope of the killer. Initialises
# the kc_kill_count variable if needed (change_variable on a character that
# has never had the variable set errors out with "Variable not of the 'value'
# scope type. Type: empty") and then increments it by 1.
kc_increment_kill_count_effect = {
	if = {
		limit = { NOT = { has_variable = kc_kill_count } }
		set_variable = {
			name = kc_kill_count
			value = 0
		}
	}
	change_variable = {
		name = kc_kill_count
		add = 1
	}
}
